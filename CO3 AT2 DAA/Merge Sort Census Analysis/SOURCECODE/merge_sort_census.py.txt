# National Census Data Sorting System using Merge Sort

def merge_sort(arr):

    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2

    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    result = []
    i = j = 0

    while i < len(left) and j < len(right):

        if left[i] < right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1

    result.extend(left[i:])
    result.extend(right[j:])

    return result


# Sample Census Region Codes
records = [38, 27, 43, 3, 9, 82, 10]

print("Original Records:", records)

sorted_records = merge_sort(records)

print("Sorted Records:", sorted_records)
